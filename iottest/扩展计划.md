## 网络传输模块（详细教程）

### 概述

网络传输模块允许ESP32通过WiFi连接到互联网，并使用MQTT协议将传感器数据上传到云端服务器。这使得您可以远程监控环境数据，实现真正的物联网应用。

创建 `src/network_module.cpp`：

```cpp
#include "network_module.h"

// 全局对象
WiFiClient espClient;
PubSubClient mqttClient(espClient);

// 网络状态
NetworkStatus netStatus = NET_DISCONNECTED;
unsigned long lastReconnectAttempt = 0;
const unsigned long RECONNECT_INTERVAL = 5000;  // 5秒重连间隔

/**
 * 初始化WiFi连接
 */
bool initWiFi() {
    Serial.println();
    Serial.println("[WiFi] 正在连接...");
    Serial.print("[WiFi] SSID: ");
    Serial.println(WIFI_SSID);
    
    netStatus = NET_WIFI_CONNECTING;
    WiFi.mode(WIFI_STA);
    WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
    
    unsigned long startAttempt = millis();
    int dots = 0;
    
    while (WiFi.status() != WL_CONNECTED) {
        delay(500);
        Serial.print(".");
        dots++;
        
        if (dots >= 40) {  // 20秒超时
            Serial.println();
            Serial.println("[WiFi] ✗ 连接失败!");
            netStatus = NET_ERROR;
            return false;
        }
        
        if (millis() - startAttempt > WIFI_TIMEOUT) {
            Serial.println();
            Serial.println("[WiFi] ✗ 连接超时!");
            netStatus = NET_ERROR;
            return false;
        }
    }
    
    Serial.println();
    Serial.println("[WiFi] ✓ 连接成功!");
    Serial.print("[WiFi] IP地址: ");
    Serial.println(WiFi.localIP());
    Serial.print("[WiFi] 信号强度: ");
    Serial.print(WiFi.RSSI());
    Serial.println(" dBm");
    
    netStatus = NET_WIFI_CONNECTED;
    return true;
}

/**
 * MQTT消息回调函数
 */
void mqttCallback(char* topic, byte* payload, unsigned int length) {
    Serial.print("[MQTT] 收到消息 [");
    Serial.print(topic);
    Serial.print("]: ");
    
    // 将payload转换为字符串
    String message;
    for (unsigned int i = 0; i < length; i++) {
        message += (char)payload[i];
    }
    Serial.println(message);
    
    // 解析JSON命令
    // 可以在这里处理来自云端的控制命令
    // 例如：{"command":"irrigate","duration":5000}
}

/**
 * 初始化MQTT连接
 */
bool initMQTT() {
    if (WiFi.status() != WL_CONNECTED) {
        Serial.println("[MQTT] ✗ WiFi未连接，无法初始化MQTT");
        return false;
    }
    
    Serial.println();
    Serial.println("[MQTT] 正在连接...");
    Serial.print("[MQTT] 服务器: ");
    Serial.print(MQTT_SERVER);
    Serial.print(":");
    Serial.println(MQTT_PORT);
    
    netStatus = NET_MQTT_CONNECTING;
    mqttClient.setServer(MQTT_SERVER, MQTT_PORT);
    mqttClient.setCallback(mqttCallback);
    mqttClient.setBufferSize(512);  // 设置MQTT缓冲区大小
    
    return reconnectMQTT();
}

/**
 * 重连MQTT服务器
 */
bool reconnectMQTT() {
    int attempts = 0;
    const int maxAttempts = 3;
    
    while (!mqttClient.connected() && attempts < maxAttempts) {
        Serial.print("[MQTT] 尝试连接 (");
        Serial.print(attempts + 1);
        Serial.print("/");
        Serial.print(maxAttempts);
        Serial.print(")... ");
        
        bool connected = false;
        
        // 根据是否有用户名密码选择连接方式
        if (strlen(MQTT_USER) > 0) {
            connected = mqttClient.connect(MQTT_CLIENT_ID, MQTT_USER, MQTT_PASSWORD);
        } else {
            connected = mqttClient.connect(MQTT_CLIENT_ID);
        }
        
        if (connected) {
            Serial.println("✓ 成功!");
            Serial.print("[MQTT] 客户端ID: ");
            Serial.println(MQTT_CLIENT_ID);
            
            // 订阅命令主题
            mqttClient.subscribe(MQTT_TOPIC_SUBSCRIBE);
            Serial.print("[MQTT] 已订阅: ");
            Serial.println(MQTT_TOPIC_SUBSCRIBE);
            
            // 发布上线状态
            StaticJsonDocument<128> statusDoc;
            statusDoc["status"] = "online";
            statusDoc["device"] = MQTT_CLIENT_ID;
            statusDoc["uptime"] = millis() / 1000;
            
            char statusBuffer[128];
            serializeJson(statusDoc, statusBuffer);
            mqttClient.publish(MQTT_TOPIC_STATUS, statusBuffer);
            
            netStatus = NET_MQTT_CONNECTED;
            return true;
        } else {
            Serial.print("✗ 失败 (rc=");
            Serial.print(mqttClient.state());
            Serial.println(")");
            attempts++;
            delay(2000);
        }
    }
    
    Serial.println("[MQTT] ✗ 连接失败!");
    netStatus = NET_ERROR;
    return false;
}

/**
 * 发布消息到MQTT主题
 */
bool publishSensorData(const char* topic, const char* payload) {
    if (!mqttClient.connected()) {
        Serial.println("[MQTT] ✗ 未连接，无法发布");
        return false;
    }
    
    bool result = mqttClient.publish(topic, payload);
    
    if (result) {
        Serial.print("[MQTT] ✓ 数据已发布到: ");
        Serial.println(topic);
    } else {
        Serial.println("[MQTT] ✗ 发布失败!");
    }
    
    return result;
}

/**
 * 发布传感器数据（JSON格式）
 */
bool publishSensorDataJSON(float temp, float humidity, int soilMoisture, 
                           uint16_t tvoc, uint16_t eco2, bool rainDetected) {
    // 创建JSON文档
    StaticJsonDocument<256> doc;
    
    doc["device"] = MQTT_CLIENT_ID;
    doc["timestamp"] = millis();
    
    JsonObject sensors = doc.createNestedObject("sensors");
    sensors["temperature"] = temp;
    sensors["humidity"] = humidity;
    sensors["soil_moisture"] = soilMoisture;
    sensors["tvoc"] = tvoc;
    sensors["eco2"] = eco2;
    sensors["rain"] = rainDetected;
    
    // 序列化JSON
    char jsonBuffer[256];
    serializeJson(doc, jsonBuffer);
    
    Serial.println("[MQTT] 发送数据:");
    Serial.println(jsonBuffer);
    
    return publishSensorData(MQTT_TOPIC_PUBLISH, jsonBuffer);
}

/**
 * 检查网络连接状态
 */
void checkNetworkConnection() {
    // 检查WiFi
    if (WiFi.status() != WL_CONNECTED) {
        Serial.println("[WiFi] 连接已断开，尝试重连...");
        netStatus = NET_DISCONNECTED;
        initWiFi();
        return;
    }
    
    // 检查MQTT
    if (!mqttClient.connected()) {
        unsigned long now = millis();
        if (now - lastReconnectAttempt > RECONNECT_INTERVAL) {
            lastReconnectAttempt = now;
            Serial.println("[MQTT] 连接已断开，尝试重连...");
            reconnectMQTT();
        }
    }
    
    // 处理MQTT消息
    if (mqttClient.connected()) {
        mqttClient.loop();
    }
}

/**
 * 获取网络状态
 */
NetworkStatus getNetworkStatus() {
    return netStatus;
}

/**
 * 获取网络状态字符串
 */
String getNetworkStatusString() {
    switch (netStatus) {
        case NET_DISCONNECTED:
            return "未连接";
        case NET_WIFI_CONNECTING:
            return "WiFi连接中...";
        case NET_WIFI_CONNECTED:
            return "WiFi已连接";
        case NET_MQTT_CONNECTING:
            return "MQTT连接中...";
        case NET_MQTT_CONNECTED:
            return "MQTT已连接";
        case NET_ERROR:
            return "连接错误";
        default:
            return "未知状态";
    }
}
```

A: 检查服务器地址和端口，确认防火墙未阻止1883端口

A: 使用MQTT over TLS (端口8883)，启用用户名密码认证

## 新增硬件组件

### 光敏传感器（光照检测）
- **用途**：检测环境光照强度
- **接口类型**：模拟ADC或I2C
- **建议型号**：BH1750（I2C，数字光照传感器）
- **连接引脚**：待定（建议GPIO21(SDA), GPIO22(SCL)使用现有I2C）
- **供电**：3.3V
- **测量范围**：0-65535 lux
- **功能**：用于自动补光控制

### LED灯（补光）
- **用途**：光照不足时自动开启补光
- **连接方式**：GPIO控制（建议GPIO23）
- **驱动方式**：可直接驱动（小功率）或通过MOS管/继电器驱动（大功率LED）
- **供电**：根据LED功率选择5V或12V独立供电
- **控制逻辑**：当光照强度低于设定阈值时自动开启

### 继电器 + 水泵（自动灌溉）
- **继电器模块**：
  - **连接引脚**：GPIO25（数字输出）
  - **供电**：3.3V（控制端） / 5V或12V（继电器模块）
  - **控制逻辑**：LOW = 继电器关断，HIGH = 继电器吸合
- **水泵**：
  - **类型**：直流微型水泵（5V或12V）
  - **连接**：通过继电器控制，独立供电
  - **注意**：继电器需加续流二极管保护，防止反向电动势损坏

## 自动控制逻辑

### 1. 自动灌溉系统
**触发条件**：
```
土壤湿度 < 设定阈值（例如：30%）
```

**执行动作**：
- 开启继电器（启动水泵）
- 灌溉持续时间：固定时间或直到土壤湿度达到目标值
- 关闭继电器（停止水泵）

**建议配置**：
```cpp
#define SOIL_MOISTURE_THRESHOLD 30    // 湿度低阈值（%）
#define IRRIGATION_DURATION 5000     // 灌溉时长（毫秒）
```

### 2. 自动补光系统
**触发条件**：
```
光照强度 < 设定阈值（例如：10000 lux）
```

**执行动作**：
- 开启LED补光灯
- 光照达到目标值或达到最大时长后关闭

**建议配置**：
```
光照阈值范围可根据实际需求调整
建议配合光周期控制（例如：仅在白天补光）
```

## 注意事项

### 电源管理
- 传感器和控制板使用同一电源时注意电流总功率
- 大功率执行器（水泵、LED）必须独立供电
- 继电器线圈两端并联续流二极管（1N4007等）

### 安全保护
- 避免水泵空转（确保水源充足）
- 设置最大灌溉时长，防止过度灌溉
- LED长时间工作注意散热
- 继电器触点加装保护电路
